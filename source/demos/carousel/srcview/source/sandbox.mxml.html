        <![CDATA[
            import away3d.materials.BitmapMaterial;
            import away3d.materials.ColorMaterial;
            import away3d.core.math.Number3D;
            
            private var carousel:CarouselGallery;
            private var gallery:Array;
            private var materials:Array;
            
            private function init():void {
                if (stage) {
                    initAway3D();
                } else {
                    this.addEventListener(Event.ADDED_TO_STAGE, function(e:Event):void { stage.quality = flash.display.StageQuality.LOW; 
                                                                                         initAway3D(); });
                }    
            }    
            
            private function initAway3D():void {
                away3dMain.view.camera.position = new Number3D(0, 0, 50);

                materials = new Array();
                gallery = new Array();
                gallery.push( "3dgallery/jack_glass.png",
                              "3dgallery/jack_glass_with_jack.png",
                              "3dgallery/studio2.png",
                              "3dgallery/studio.png",
                              "3dgallery/valentine.png",
                              "3dgallery/wine_glasses.png",
                              "3dgallery/knot.jpg",
                              "3dgallery/dispersion.avi1.jpg",
                              "3dgallery/dispersion2.avi2.jpg",
                              "3dgallery/spin.avi3.jpg");
                for (var i:int = 0; i < gallery.length; i++) {
                    var request:URLRequest = new URLRequest(gallery[i]);
                    var imageLoader:Loader = new Loader();
                    imageLoader.contentLoaderInfo.addEventListener(Event.COMPLETE, imageLoaded);
                    imageLoader.load(request);
                }            
                
                this.addEventListener(Event.ENTER_FRAME, render);
            }
            
            private function imageLoaded(event:Event):void{
                var bm:BitmapMaterial = new BitmapMaterial(Bitmap(event.currentTarget.content).bitmapData);
                bm.smooth = true;
                bm.alpha = 1;
                materials.push(bm);      
                
                if (materials.length == gallery.length) {
                    loadingMessage.visible = false;
                    carousel = new CarouselGallery({planeHeight:20, planeWidth:20, planeSpacing:10, galleryMaterials:materials});
                    carousel.position = new Number3D(0,0,0);
                    away3dMain.view.scene.addChild(carousel);    
                }
            }
            
            private function changeSpacing(event:Event):void {
                if (carousel) {
                    carousel.planeSpacing = spacingSlider.value;
                }
            }
            
            private function changeHeight(event:Event):void {
                if (carousel) {
                    carousel.planeHeight = heightSlider.value;
                }
            }
            
            private function changeWidth(event:Event):void {
                if (carousel) {
                    carousel.planeWidth = widthSlider.value;
                }
            }
            
            private function render(evt:Event):void {
                if (carousel) {
                    carousel.rotationY += .2;
                }
                away3dMain.view.render();
            }
