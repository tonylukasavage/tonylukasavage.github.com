<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" layout="absolute" xmlns:local="*"
                backgroundColor="0x000000" creationComplete="init();" frameRate="60" height="600" width="800" viewSourceURL="srcview/index.html">
    <mx:Script>
        <![CDATA[
            import mx.core.UIComponent;
            
            private var _camera:Camera;
            private var _video:Video;
            
            private function init():void {
                if (stage) {
                    initMain();
                } else {
                    this.addEventListener(Event.ADDED_TO_STAGE, function(e:Event):void { initMain(); });
                }    
            }    
            
            private function initMain():void {
                stage.quality = flash.display.StageQuality.HIGH;
                _camera = Camera.getCamera();
                _camera.setQuality(0,100);
                _camera.setMode(800,600,60,true);
                
                var ui:UIComponent = new UIComponent();
                _video = new Video(800,600);
                _video.attachCamera(_camera);
                ui.addChild(_video);
                
                this.addChild(ui);
                this.setChildIndex(ui, 0);
                this.addEventListener(Event.ENTER_FRAME, _onEnterFrame);
            }
            
            private function _onEnterFrame(event:Event):void {
                controlPanel.title = "FPS: " + _camera.currentFPS.toString() + " / " + _camera.fps.toString();
            }

            private function _changeQuality():void {
                if (_camera) {
                    _camera.setQuality(0, qualitySlider.value);
                }
            }
            
            private function _changeMode(type:String):void {
                if (_camera) {
                    if (type == "size") {
                        var h:uint = 600 * sizeSlider.value;
                        var w:uint = 800 * sizeSlider.value;
                        _camera.setMode(w, h, 60, true);
                        _video.height = h;
                        _video.width = w; 
                    } else if (type == "fps") {
                        _camera.setMode(_camera.width, _camera.height, fpsSlider.value, true);
                    }
                }
            }
            
            private function _onMouseDown(event:MouseEvent):void {
                controlPanel.startDrag();
            }
            
            private function _onMouseUp(event:MouseEvent):void {
                controlPanel.stopDrag();
            }
        ]]>
    </mx:Script>
    <mx:Panel x="10" y="416" width="265" height="174" layout="absolute" id="controlPanel" title="Current FPS: " 
              mouseDown="_onMouseDown(event);" mouseUp="_onMouseUp(event);">
        <mx:HSlider x="64" y="10" minimum="1" maximum="100" dataTipPlacement="top" tickColor="#000000" snapInterval="10" tickInterval="10"
                    labels="[1,50,100]" liveDragging="false" id="qualitySlider" value="50" change="_changeQuality();" width="169"/>
        <mx:HSlider x="64" y="47" minimum="0" maximum="1" dataTipPlacement="top" tickColor="#000000" snapInterval=".1" tickInterval=".1"
                    labels="[1,50,100]" liveDragging="false" id="sizeSlider" value="1" change="_changeMode('size');" width="169"/>
        <mx:HSlider x="64" y="80" minimum="0" maximum="24" dataTipPlacement="top" tickColor="#000000" snapInterval="4" tickInterval="4"
                    labels="[0,12,24]" liveDragging="false" id="fpsSlider" value="12" change="_changeMode('fps');" width="169"/>
        <mx:Label x="10" y="22" text="Quality:" width="67" fontWeight="bold"/>
        <mx:Label x="10" y="58" text="Size(%):" width="67" fontWeight="bold"/>
        <mx:Label x="10" y="91" text="FPS:" width="55" fontWeight="bold"/>
    </mx:Panel>
    
</mx:Application>
