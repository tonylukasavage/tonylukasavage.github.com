package
{
    import away3d.cameras.HoverCamera3D;
    import away3d.containers.View3D;
    import away3d.core.utils.Cast;
    import away3d.debug.AwayStats;
    import away3d.materials.BitmapMaterial;
    import away3d.materials.TransformBitmapMaterial;
    import away3d.primitives.Sphere;
    
    import com.greensock.*;
    import com.greensock.easing.*;
    
    import flash.display.Bitmap;
    import flash.display.BitmapData;
    import flash.display.DisplayObject;
    import flash.display.DisplayObjectContainer;
    import flash.display.Sprite;
    import flash.display.StageQuality;
    import flash.events.Event;
    import flash.events.MouseEvent;
    import flash.geom.Matrix;
    import flash.geom.Rectangle;
    import flash.geom.Vector3D;
    
    [SWF("600", width="800", backgroundColor="0x000000", frameRate="60")]
    public class away3d_motion extends Sprite
    {
        private const NUMBER_OF_BITMAPS:uint = 50;
        private var _view:View3D;
        private var _view2:View3D;
        private var _camera:HoverCamera3D;
        private var _sphere:Sphere;
        private var _sphere2:Sphere;
        private var _bigSphere:Sphere;
        private var _drawn:DisplayObject;
        private var _bounds:Rectangle;
        private var _bmd:BitmapData;
        private var _matrix:Matrix;
        
        [Embed("ice.jpg")] private var iceImage:Class;
        [Embed("lava.jpg")] private var lavaImage:Class;
        [Embed("starfield2.jpg")] private var starfieldImage:Class;
        
        public function away3d_motion()
        {
            if (stage) {
                init();
            } else {
                this.addEventListener(Event.ADDED_TO_STAGE, function(e:Event):void { init(); });
            }
        }
        
        private function init():void {
            _matrix = new Matrix();
            stage.quality = StageQuality.LOW;
            
            // configure view
            _camera = new HoverCamera3D();
            _view = new View3D({camera:_camera});
            _view.x = stage.stageWidth/2;
            _view.y = stage.stageHeight/2;
            _camera.distance = 1000;
            _camera.panAngle = 0;
            _camera.tiltAngle = 0;
            _camera.hover(true);
            
            // configure view2
            _view2 = new View3D();
            _view2.x = stage.stageWidth/2;
            _view2.y = stage.stageHeight/2;
            
            // configure sphere
            _sphere = new Sphere({material:new BitmapMaterial(Cast.bitmap(iceImage))});
            _sphere.x = -300;
            _sphere.y = -100;
            _view.scene.addChild(_sphere);
            TweenMax.to(_sphere, 1.5, {x:300, y:100, ease:Cubic.easeInOut, repeat:-1, yoyo:true});
            
            // configure sphere
            _sphere2 = new Sphere({material:new BitmapMaterial(Cast.bitmap(lavaImage))});
            _sphere2.x = 250;
            _sphere2.z = 200;
            _view.scene.addChild(_sphere2);
            TweenMax.to(_sphere2, 0.75, {x:-250, z:-200, ease:Cubic.easeInOut, repeat:-1, yoyo:true});
            
            // create starfield sphere
            _bigSphere = new Sphere({radius:10000, material:new TransformBitmapMaterial(Cast.bitmap(starfieldImage),{repeat:true, scaleX:0.33, scaleY:0.33})});
            _bigSphere.scaleX = -1;
            _view2.scene.addChild(_bigSphere);
            
            // add elements to scene
            this.addChild(_view2);
            this.addChild(_view);
            this.addChild(new AwayStats());
            
            this.addEventListener(Event.ENTER_FRAME, onEnterFrame);
            //this.stage.addEventListener(MouseEvent.MOUSE_MOVE, onMouseMove);
        }
        
        private function onMouseMove(e:MouseEvent):void {
            _camera.panAngle = (e.stageX-(stage.stageWidth/2)) * 90 / (stage.stageWidth/2);
            _camera.tiltAngle = (e.stageY-(stage.stageHeight/2)) * 30 / (stage.stageHeight/2);
        }
        
        private function onEnterFrame(e:Event):void {
            var bitmap:Bitmap;
            
            _bigSphere.rotationY += 0.2;
            _camera.panAngle += 1;
            _camera.hover();
            _view2.render();
            _view.render();
            
            _drawn = (_view.session.getContainer(_view) as DisplayObjectContainer).getChildAt(0);
            _bounds = _drawn.getRect(this);
            _matrix.createBox(1, 1, 0, _view.width/2 - _bounds.x, _view.height/2 - _bounds.y);
            _bmd = new BitmapData(_bounds.width, _bounds.height, true, 0x00ffffff);
            _bmd.draw(_drawn, _matrix);
            
            bitmap = new Bitmap(_bmd);
            bitmap.alpha = 0.5;
            bitmap.cacheAsBitmap = true;
            this.addChild(bitmap);
            TweenLite.to(bitmap, 0.5, {alpha:0, ease:Linear.easeNone, onCompleteParams:[this, bitmap], 
                onComplete:function(layer:Sprite, bm:Bitmap):void {
                    layer.removeChild(bm);
                    bm = null;
                }
            });
            bitmap.x = _bounds.x;
            bitmap.y = _bounds.y;
        }
    }
}
