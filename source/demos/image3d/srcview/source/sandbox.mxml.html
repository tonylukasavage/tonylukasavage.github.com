<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" layout="absolute" xmlns:local="*"
                backgroundColor="0x888888" creationComplete="init();" frameRate="60" height="600" width="800" viewSourceURL="srcview/index.html">
    <mx:Script>
        <![CDATA[
            import mx.events.ColorPickerEvent;
            import away3dlite.materials.ColorMaterial;
            import away3dlite.primitives.Cube6;
            import away3dlite.cameras.Camera3D;
            import away3dlite.core.utils.Cast;
            import flash.filters.GlowFilter;
            
            [Embed(source="wallpaper.jpg")] private var MyImage:Class;
            private var CHUNK:uint = 50;
            
            private var isLeftButtonDown:Boolean = false;
            private var lastX:Number, lastY:Number;
            private var angleX:Number = Math.PI/2;
            private var angleY:Number = Math.PI/2;
            private var radius:Number;
        
            private function init():void {
                if (stage) {
                    initAway3D();
                } else {
                    this.addEventListener(Event.ADDED_TO_STAGE, function(e:Event):void { stage.quality = flash.display.StageQuality.LOW; 
                                                                                         initAway3D(); });
                }    
                
                
            }    
            
            private function initAway3D():void {
                radius = away3dMain.view.camera.z = -5000 * (50 / CHUNK);
                away3dMain.view.camera.lookAt(new Vector3D(0,0,0));    
                handleImage();
                
                stage.addEventListener(MouseEvent.MOUSE_DOWN, function(e:MouseEvent):void { isLeftButtonDown = true; lastX = e.stageX; lastY = e.stageY; } );
                stage.addEventListener(MouseEvent.MOUSE_UP, function(e:MouseEvent):void { isLeftButtonDown = false; } );
                stage.addEventListener(MouseEvent.MOUSE_MOVE, doZoom);
                updateCameraPosition();
                this.addEventListener(Event.ENTER_FRAME, render);
            }
            
            private function doZoom(event:MouseEvent):void {
                if (isLeftButtonDown) {
                    if (event.shiftKey) {
                        radius += (event.stageY - lastY) * 50;
                    } else {
                        angleY += (event.stageY - lastY) / 100;
                        angleX += (event.stageX - lastX) / 100;         
                    }
                    updateCameraPosition();
                    lastX = event.stageX;
                    lastY = event.stageY;
                }
            }
            
            private function updateCameraPosition():void {
                away3dMain.view.camera.x = radius * Math.cos(angleX) * Math.sin(angleY);
                away3dMain.view.camera.y = radius * Math.cos(angleY);
                away3dMain.view.camera.z = radius * Math.sin(angleX) * Math.sin(angleY);
                away3dMain.view.camera.lookAt(new Vector3D(0, 0, 0), Math.sin(angleY) < 0 ? new Vector3D(0,1,0) : new Vector3D(0,-1,0));
            }
            
            private function handleImage():void {
                var data:BitmapData = Cast.bitmap(MyImage);
                var widthNum:uint = data.width / CHUNK;
                var heightNum:uint = data.height / CHUNK;
                var cubeSize:uint = 100;
                var cubeSpace:Number = cubeSize * 1.3;
                var startX:Number = cubeSize/2 - (widthNum/2*cubeSpace);
                var startY:Number = cubeSize/2 - (heightNum/2*cubeSpace);
                
                for (var h:int = 0; h < heightNum; h++) {
                    for (var w:int = 0; w < widthNum; w++) {
                        var color:uint = getColorAverage(w * CHUNK, h * CHUNK, data);
                        var cube:Cube6 = new Cube6(new ColorMaterial(color), cubeSize, cubeSize, cubeSize);
                        cube.x = startX + w * cubeSpace;
                        cube.y = startY + h * cubeSpace;
                        cube.z = 0;
                        cube.mouseEnabled = false;
                        cube.layer = new Sprite();
                        away3dMain.view.addChild(cube.layer);
                        cube.layer.blendMode = BlendMode.NORMAL;
                        away3dMain.view.scene.addChild(cube);
                    }
                }
            }
            
            private function getColorAverage(x:int, y:int, data:BitmapData):uint {
                var color:uint;
                var red:uint = 0;
                var green:uint = 0;
                var blue:uint = 0;
                var div:uint = CHUNK * CHUNK;
    
                for (var i:uint = 0; i < CHUNK; i++) {
                    for (var j:uint = 0; j < CHUNK; j++) {
                        color = data.getPixel(x + i, y + j);
                        red += color >> 16;
                        green += (color >> 8) & 0xff;
                        blue += color & 0x00ff;
                    }
                }
                return ((red/div) << 16) | ((green/div) << 8) | (blue/div);
            }
            
            private function changeBlendMode(event:Event):void {
                for (var i:uint = 0; i < away3dMain.view.scene.children.length; i++) {
                    (away3dMain.view.scene.getChildAt(i) as Cube6).layer.blendMode = event.currentTarget.selectedItem.data;
                }
            }
            
            private function changeBackgroundColor(event:ColorPickerEvent):void {
                mx.core.Application.application.setStyle("backgroundColor", event.color);
            }
            
            private function render(evt:Event):void {
                if (away3dMain.doRender) { away3dMain.view.render(); }
            }
        ]]>
    </mx:Script>
