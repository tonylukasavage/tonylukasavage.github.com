package
{
    import away3dlite.containers.ObjectContainer3D;
    import away3dlite.containers.Scene3D;
    import away3dlite.containers.View3D;
    import away3dlite.core.utils.Cast;
    import away3dlite.events.Loader3DEvent;
    import away3dlite.loaders.Loader3D;
    import away3dlite.loaders.Max3DS;
    import away3dlite.materials.BitmapMaterial;
    
    import com.transmote.flar.FLARManager;
    import com.transmote.flar.marker.FLARMarker;
    import com.transmote.flar.utils.geom.FLARGeomUtils;
    
    import flash.display.Sprite;
    import flash.display.Stage;
    import flash.events.Event;
    
    import org.libspark.flartoolkit.core.param.FLARParam;
    import org.libspark.flartoolkit.support.away3d_lite.FLARCamera3D;

    public class Away3DLiteSprite extends Sprite
    {
        private var _mirror:Boolean;
        private var _marker:FLARMarker = null;
        private var view:View3D;
        private var model:ObjectContainer3D = null;
        private var container:ObjectContainer3D;
        private var loader:Loader3D;
        
        [Embed(source="resources/Cmro_tex.png")] private var camaroTexture:Class;
        
        public function Away3DLiteSprite(flarManager:FLARManager, viewportWidth:Number, viewportHeight:Number, stage:Stage)
        {
            _mirror = flarManager.mirrorDisplay;
            this.initAway3DEnvironment(flarManager.cameraParams, viewportWidth, viewportHeight, stage,
                                       viewportWidth/flarManager.flarSource.sourceSize.width);
        }
        
        private function initAway3DEnvironment(cameraParams:FLARParam, viewportWidth:Number, viewportHeight:Number, stage:Stage, viewportToSourceWidthRatio:Number) :void {
            this.view = new View3D(new Scene3D(), new FLARCamera3D(cameraParams, viewportToSourceWidthRatio));
            this.view.x = 0.5 * viewportWidth;
            this.view.y = 0.5 * viewportHeight;
            this.view.z = 0;
            this.addChild(this.view);
            
            container = new ObjectContainer3D();
            view.scene.addChild(container);
            
            var parser:Max3DS = new Max3DS();
            parser.material = new BitmapMaterial(Cast.bitmap(camaroTexture));
            parser.scaling = .5;
            loader = new Loader3D();
            loader.addEventListener(Loader3DEvent.LOAD_SUCCESS, onLoadSuccess);
            loader.loadGeometry("resources/cmro.3DS", parser);
            
            this.addEventListener(Event.ENTER_FRAME, this.onEnterFrame);
        }
        
        public function addMarker(marker:FLARMarker):void {
            _marker = marker;
            if (model) { 
                container.addChild(model); 
            }
        }
        
        private function onLoadSuccess(event:Loader3DEvent):void {
            model = loader.handle as ObjectContainer3D;    
            model.rotationX = -90;
            model.rotationZ = -90;
            model.rotationY = 90;
        }
        
        public function removeMarker(marker:FLARMarker) :void {
            if (model && container.contains(model)) { 
                container.removeChild(model); 
            }
            _marker = null;
        } 
        
        private function onEnterFrame (evt:Event) :void {
            if (_marker && model) {
                container.transform.matrix3D = FLARGeomUtils.convertFLARMatrixToFlashMatrix3D(_marker.transformMatrix, _mirror);
            }
            this.view.render();
        }
    }
}
