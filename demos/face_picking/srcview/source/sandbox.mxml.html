<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" layout="absolute" xmlns:local="*"
                creationComplete="init();" height="600" width="800" frameRate="60" backgroundColor="0xdddddd" viewSourceURL="srcview/index.html">
    <mx:Script>
        <![CDATA[
            import away3d.core.draw.ScreenVertex;
            import away3d.core.base.Vertex;
            import away3d.materials.WireColorMaterial;
            import away3d.core.draw.DrawTriangle;
            import away3d.events.MouseEvent3D;
            import away3d.core.base.Face;
            import away3d.primitives.Sphere;
            import away3d.core.math.Number3D;
            
            private function init():void {
                if (stage) {
                    initAway3D();
                } else {
                    this.addEventListener(Event.ADDED_TO_STAGE, function(e:Event):void { initAway3D(); });
                }    
            }    
            
            private function initAway3D():void {
                initScene();
                initObjects();
                this.addEventListener(Event.ENTER_FRAME, render);    
            }
            
            private function initScene():void {
                away3dMain.view.camera.position = new Number3D(0, 0, -450);
                away3dMain.view.camera.lookAt(new Number3D(0,0,0));    
            }
            
            private function initObjects():void {
                var sphere:Sphere = new Sphere({material:new WireColorMaterial(0x000000,{wireColor:0xffffff}),
                                                segmentsH:20, 
                                                segmentsW:20});
                sphere.useHandCursor = true;
                sphere.addEventListener(MouseEvent3D.MOUSE_DOWN, onMouseDown);
                away3dMain.view.scene.addChild(sphere);    
            }
            
            private function onMouseDown(event:MouseEvent3D):void {
                var sphere:Sphere = event.target as Sphere;
                if (event.drawpri is DrawTriangle) {
                    var face:Face = (event.drawpri as DrawTriangle).faceVO.face as Face;
                    faceNumber.text = "Face Number: " + String((event.target as Sphere).faces.indexOf(face));
                    vertex0.text = "vertex0: " + vertexToString(face.v0);
                    vertex1.text = "vertex1: " + vertexToString(face.v1);
                    vertex2.text = "vertex2: " + vertexToString(face.v2);
                    screen0.text = "screen0: " + svToString(away3dMain.camera.screen(sphere, face.v0)); 
                    screen1.text = "screen1: " + svToString(away3dMain.camera.screen(sphere, face.v1)); 
                    screen2.text = "screen2: " + svToString(away3dMain.camera.screen(sphere, face.v2)); 

                    if (face.material is WireColorMaterial) {
                        face.material = null;
                    } else {
                        face.material = new WireColorMaterial(0xff0000,{wireColor:0xffffff});
                    }
                }
            }
            
            private function vertexToString(vertex:Vertex):String {
                return "(" + String(roundTo2Decimals(vertex.x)) + "," +
                             String(roundTo2Decimals(vertex.y)) + "," +
                             String(roundTo2Decimals(vertex.z)) + ")";
            }
            
            private function svToString(sv:ScreenVertex):String {
                return "(" + String(roundTo2Decimals(sv.x+away3dMain.view.x)) + "," +
                             String(roundTo2Decimals(sv.y+away3dMain.view.y)) + ")";
            }
            
            private function roundTo2Decimals(val:Number):Number {
                return Math.round(val*100)/100;
            }
            
            private function render(evt:Event):void {
                away3dMain.view.render();
            }
        ]]>
    </mx:Script>
