    <mx:Script>
        <![CDATA[
            import away3d.materials.AnimatedBitmapMaterial;
            import away3d.materials.VideoMaterial;
            import away3d.materials.PhongColorMaterial;
            import away3d.core.utils.Color;
            import away3d.materials.ShadingColorMaterial;
            import away3d.materials.Dot3BitmapMaterialF10;
            import away3d.materials.WireframeMaterial;
            import away3d.materials.WireColorMaterial;
            import away3d.materials.ColorMaterial;
            import away3d.lights.DirectionalLight3D;
            import away3d.materials.Dot3BitmapMaterial;
            import away3d.primitives.Sphere;
            import away3d.core.math.Number3D;
            import away3d.core.utils.Cast;
            
            [Embed(source="resources/normal.jpg")] private var SphereNormalImage:Class;
            [Embed(source="resources/tvnoise.swf", mimeType="application/octet-stream")] public var TvNoiseSwf:Class;

            private var light:DirectionalLight3D;
            
            private var sphere:Sphere;
            private var animatedBitmapMaterial:AnimatedBitmapMaterial;
            private var dot3Material:Dot3BitmapMaterial
            private var dot3F10Material:Dot3BitmapMaterialF10;
            private var colorMaterial:ColorMaterial;
            private var phongColorMaterial:PhongColorMaterial;
            private var shadingColorMaterial:ShadingColorMaterial;
            private var videoMaterial:VideoMaterial;
            private var wireColorMaterial:WireColorMaterial;
            private var wireframeMaterial:WireframeMaterial;
            private var loader:Loader;
            
            private function init():void {
                if (stage) {
                    initAway3D();
                } else {
                    this.addEventListener(Event.ADDED_TO_STAGE, function(e:Event):void { initAway3D(); });
                }    
            }    
            
            private function initAway3D():void {
                initScene();
                initObjects();
                initMaterials();
                
                this.addEventListener(Event.ENTER_FRAME, render);    
            }
            
            private function initScene():void {
                light = new DirectionalLight3D();
                light.direction = new Number3D(0,0,1);
                
                away3dMain.view.camera.position = new Number3D(100, 100, -350);
                away3dMain.view.camera.lookAt(new Number3D(0,0,0));    
                away3dMain.view.scene.addLight(light);
            }
            
            private function initMaterials():void {
                var baseColor:int = 0x0000ff;
                dot3F10Material = new Dot3BitmapMaterialF10(Cast.bitmap(new SphereNormalImage()), Cast.bitmap(new SphereNormalImage()));
                dot3Material = new Dot3BitmapMaterial(new BitmapData(1, 1, false, baseColor), Cast.bitmap(new SphereNormalImage()));    
                colorMaterial = new ColorMaterial(baseColor);
                phongColorMaterial = new PhongColorMaterial(baseColor);
                shadingColorMaterial = new ShadingColorMaterial(baseColor);
                videoMaterial = new VideoMaterial({file:"resources/tvnoise.flv", loop:true});
                wireColorMaterial = new WireColorMaterial(baseColor,{wireColor:0xffffff});
                wireframeMaterial = new WireframeMaterial(baseColor);
                
                loader = new Loader(); 
                loader.contentLoaderInfo.addEventListener(Event.INIT, loadedAnimatedBitmapMaterial); 
                loader.loadBytes(new TvNoiseSwf()); 
            }
            
            private function loadedAnimatedBitmapMaterial(event:Event):void {
                var mc:MovieClip = MovieClip(loader.content); 
                animatedBitmapMaterial = new AnimatedBitmapMaterial(mc); 
                animatedBitmapMaterial.play(); 
            }
            
            private function initObjects():void {
                sphere = new Sphere({segmentsW:5, segmentsH:5, radius:50});
                away3dMain.view.scene.addChild(sphere);    
            }

            private function changeSphereSegments():void {
                sphere.segmentsH = segmentSlider.value;
                sphere.segmentsW = segmentSlider.value;
            }
            
            private function changeMaterial(event:Event):void {
                switch (event.currentTarget.selectedItem) {
                    case "AnimatedBitmapMaterial":
                        sphere.material = animatedBitmapMaterial;
                        break;
                    case "Dot3BitmapMaterialF10":
                        sphere.material = dot3F10Material;
                        break;
                    case "Dot3BitmapMaterial":
                        sphere.material = dot3Material;
                        break;
                    case "ColorMaterial":
                        sphere.material = colorMaterial;
                        break;
                    case "PhongColorMaterial":
                        sphere.material = phongColorMaterial;
                        break;
                    case "ShadingColorMaterial":
                        sphere.material = shadingColorMaterial;
                        break;
                    case "VideoMaterial":
                        sphere.material = videoMaterial;
                        break;
                    case "WireColorMaterial":
                        sphere.material = wireColorMaterial;
                        break;
                    case "WireframeMaterial":
                        sphere.material = wireframeMaterial;
                        break;
                }
            }
            
            private function render(evt:Event):void {
                sphere.rotationY += 0.5;
                away3dMain.view.render();
            }
