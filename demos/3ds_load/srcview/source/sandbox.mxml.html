<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" layout="absolute" xmlns:local="*"
                backgroundColor="0xdddddd" creationComplete="init();" frameRate="60" height="600" width="800" viewSourceURL="srcview/index.html">
    <mx:Script>
        <![CDATA[
            import away3d.materials.WireColorMaterial;
            import away3d.loaders.Loader3D;
            import away3d.core.math.Number3D;
            import away3d.loaders.Max3DS;
            
            private var model:Loader3D;
            
            private function init():void {
                if (stage) {
                    initAway3D();
                } else {
                    this.addEventListener(Event.ADDED_TO_STAGE, function(e:Event):void { stage.quality = flash.display.StageQuality.LOW; 
                                                                                         initAway3D(); });
                }    
            }    
            
            private function initAway3D():void {
                away3dMain.view.camera.position = new Number3D(0, 0, -100);
                away3dMain.view.camera.lookAt(new Number3D(0,0,0));    
                
                model = Max3DS.load("models/tk.3DS", {material:new WireColorMaterial(0x000000,{wirecolor:0xffffff})});
                away3dMain.view.scene.addChild(model);
                model.addOnSuccess(onSuccess);    
                this.addEventListener(Event.ENTER_FRAME, render);
            }
            
            private function onSuccess(event:Event):void {
                model.handle.rotationX = 90;
            }
            
            private function render(evt:Event):void {
                model.handle.rotationY += .5;
                away3dMain.view.render();
            }
        ]]>
    </mx:Script>
    <local:AwayUIC id="away3dMain"  x="0" y="0" height="600" width="800"/>
</mx:Application>
