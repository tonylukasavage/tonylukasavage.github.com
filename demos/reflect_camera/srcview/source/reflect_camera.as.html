package {
    import __AS3__.vec.Vector;
    
    import away3d.cameras.HoverCamera3D;
    import away3d.cameras.TargetCamera3D;
    import away3d.containers.View3D;
    import away3d.core.math.Number3D;
    import away3d.core.utils.Cast;
    import away3d.debug.AwayStats;
    import away3d.materials.BitmapMaterial;
    import away3d.materials.ColorMaterial;
    import away3d.primitives.Cube;
    import away3d.primitives.Plane;
    import away3d.primitives.Sphere;
    
    import flash.display.*;
    import flash.events.Event;
    import flash.events.MouseEvent;
    import flash.geom.Matrix;

    [SWF(backgroundColor="#ffffff", width="800", height="600", frameRate="60")]
    public class reflect_camera extends Sprite
    {
        private var _view:View3D;
        private var _plane:Plane;
        private var _sphere:Sphere;
        private var _cube:Cube;
        private var _reflectPlane:Plane;
        private var _reflectPlane2:Plane;
        
        private var _rView:View3D;
        private var _rSphere:Sphere;
        
        private var _rView2:View3D;
        private var _rCube:Cube;
        private var _views:Vector.<View3D> = new Vector.<View3D>();
        private var _cubes:Vector.<Cube> = new Vector.<Cube>();
        private var _reflects:Vector.<Plane> = new Vector.<Plane>();
        
        private var _drawn:DisplayObject;
        private var _matrix:Matrix = new Matrix();
        private var _bmd:BitmapData;
        private var _drawnHeight:Number, _drawnWidth:Number, _drawnHeight2:Number, _drawnWidth2:Number;
        private var _reflectAlpha:Number = 0.4;
        private var _angle:Number = 0;
        private var _dirY:Boolean = true;
        
        [Embed(source="earth.jpg")] private var earthImage:Class;
        [Embed(source="wood.jpg")] private var woodImage:Class;
        
        public function reflect_camera()
        {
            if (stage) init();
            else addEventListener(Event.ADDED_TO_STAGE, init);
        }
        
        private function init(e:Event = null):void 
        {
            removeEventListener(Event.ADDED_TO_STAGE, init);
            
            // configure stage
            stage.scaleMode = StageScaleMode.NO_SCALE;
            stage.align = StageAlign.TOP_LEFT;
            stage.quality = StageQuality.HIGH;
            
            // set background
            var bg:Sprite = new Sprite();
            var matrix:Matrix = new Matrix();
            matrix.createGradientBox(stage.stageWidth, stage.stageHeight, Math.PI/2);
            bg.graphics.beginGradientFill("linear", [0x888888,0xffffff,0x888888], [1,1,1], [0,127,255], matrix);
            bg.graphics.drawRect(0,0,stage.stageWidth,stage.stageHeight);
            bg.graphics.endFill();
            this.addChild(bg);
        
            // create shared material
            var material:BitmapMaterial = new BitmapMaterial(Cast.bitmap(earthImage));
            
            // configure main away3d view
            _view = new View3D({camera:new HoverCamera3D()});
            _view.x = stage.stageWidth / 2;
            _view.y = stage.stageHeight / 2;
            _view.camera.position = new Number3D(0,500,-2500);
            _view.camera.lookAt(new Number3D(0,0,0));
            _plane = new Plane({height:1000, width:1000, material:new ColorMaterial(0x000000)});
            _plane.ownCanvas = true;
            _plane.pushback = true;
            _sphere = new Sphere({y:200, material:material});
            _cube = new Cube({y:100, material:new BitmapMaterial(Cast.bitmap(woodImage))});
            _view.camera.zoom = 5;
            (_view.camera as HoverCamera3D).minTiltAngle = 0;
            _view.scene.addChild(_plane);
            _view.scene.addChild(_sphere);
            _view.scene.addChild(_cube);

            // configure away3d reflection views
            _rView = new View3D({camera:new TargetCamera3D()});
            _rView.x = _view.x;
            _rView.y = _view.y;
            _rSphere = new Sphere({y:200, material:material});
            _rView.camera.position = new Number3D(_rSphere.x, _rSphere.y-_rSphere.radius, _rSphere.z);
            (_rView.camera as TargetCamera3D).target = _rSphere; 
            (_rView.camera as TargetCamera3D).zoom = 2;
            (_rView.camera as TargetCamera3D).focus = 100;
            _rView.scene.addChild(_rSphere);
            _rView.render();
            
            /*
            for (var i:uint = 0; i < 5; i++) {
                var view:View3D = new View3D({camera:new TargetCamera3D()});
                view.x = _view.x;
                view.y = _view.y;
                var cube:Cube = new Cube({y:100, z:i*150-300, material:new BitmapMaterial(Cast.bitmap(woodImage))});
                view.camera.position = new Number3D(cube.x, cube.y-cube.height, cube.z);
                (view.camera as TargetCamera3D).target = cube; 
                (view.camera as TargetCamera3D).zoom = 2;
                (view.camera as TargetCamera3D).focus = 100;
                view.scene.addChild(cube);
                view.render();
                
                var reflect:Plane = new Plane({height:cube.height, width:cube.width, 
                                               material:new BitmapMaterial(new BitmapData(1,1,true,0),{smooth:true, alpha:_reflectAlpha})});
                reflect.ownCanvas = true;
                _view.scene.addChild(reflect);
                
                _views.push(view);
                _cubes.push(cube);
                _reflects.push(reflect);
            }
            */
            
            
            _rView2 = new View3D({camera:new TargetCamera3D()});
            _rView2.x = _view.x;
            _rView2.y = _view.y;
            _rCube = new Cube({y:100, material:new BitmapMaterial(Cast.bitmap(woodImage))});
            _rView2.camera.position = new Number3D(_rCube.x, _rCube.y-_rCube.height, _rCube.z);
            (_rView2.camera as TargetCamera3D).target = _rCube; 
            (_rView2.camera as TargetCamera3D).zoom = 2;
            (_rView2.camera as TargetCamera3D).focus = 100;
            _rView2.scene.addChild(_rCube);
            _rView2.render();
            
            
            // create reflection plane
            _reflectPlane = new Plane({height:_sphere.radius*2, width:_sphere.radius*2, 
                                       material:new BitmapMaterial(new BitmapData(1,1,true,0),{smooth:true, alpha:_reflectAlpha})});
            _reflectPlane.ownCanvas = true;
            _view.scene.addChild(_reflectPlane);
            
            _reflectPlane2 = new Plane({height:_cube.height, width:_cube.width, 
                                        material:new BitmapMaterial(new BitmapData(1,1,true,0),{smooth:true, alpha:_reflectAlpha})});
            _reflectPlane2.ownCanvas = true;
            _view.scene.addChild(_reflectPlane2);
            
            
            this.addChild(_view);
            this.addChild(new AwayStats());
            
            this.addEventListener(Event.ENTER_FRAME, _onEnterFrame);
            stage.addEventListener(MouseEvent.MOUSE_MOVE, _onMouseMove);
        }
        
        private function _onMouseMove(e:MouseEvent):void {
            (_view.camera as HoverCamera3D).panAngle = -e.stageX/stage.stageWidth * 360;
            (_view.camera as HoverCamera3D).tiltAngle = (stage.stageHeight - e.stageY) * (90/stage.stageHeight);
        }
        
        private function _onEnterFrame(e:Event):void {
            (_view.camera as HoverCamera3D).hover();
            _angle += 0.03;
            
            // move sphere
            _sphere.x = Math.cos(_angle) * 300;
            _sphere.z = Math.sin(_angle) * 300;
            if (_dirY) {
                if (_sphere.y < 1500) {
                    _sphere.y += 2;
                } else {
                    _dirY = false;
                    _sphere.y -= 2;
                }
            } else {
                if (_sphere.y > 100) {
                    _sphere.y -= 2;
                } else {
                    _dirY = true;
                    _sphere.y += 2;
                }
            }
            _sphere.rotationY += 1;
            _sphere.rotationX += 0.5;
            
            // move cube
            
            
            _cube.x = Math.cos(_angle) * 300;
            _cube.y = (Math.sin(_angle) + 1) * 300 + 100;
            _cube.rotationZ += 2;
            
            // make reflection sphere mimic main sphere
            _rSphere.position = _sphere.position;
            _rSphere.rotationY = _sphere.rotationY;
            _rSphere.rotationX = _sphere.rotationX;
            
            /*
            for (var i:uint = 0; i < 5; i++) {
                var cube:Cube = _cubes[i];
                cube.position = 
            }
            */
            
            // do again
            _rCube.position = _cube.position;
            _rCube.rotationZ = _cube.rotationZ;
            
            // reposition and resize reflection planes
            _rView.camera.position = new Number3D(_rSphere.x, _rSphere.y-_rSphere.radius, _rSphere.z);
            _reflectPlane.position = new Number3D(_rSphere.x, 0, _rSphere.z);
            _reflectPlane.scale(1 - (_sphere.y/2000) > 0 ? 1 - (_sphere.y/2000) : 0);
            
            _rView2.camera.position = new Number3D(_rCube.x, _rCube.y-_rCube.height, _rCube.z);
            _reflectPlane2.position = new Number3D(_rCube.x, 0, _rCube.z);
            _reflectPlane2.scale(1 - (_cube.y/2000) > 0 ? 1 - (_cube.y/2000) : 0);
            
            // render reflection views
            _rView.render();
            _rView2.render();
            
            // find 2d bounding box of sphere and redraw reflection
            _drawn = (_rView.session.getContainer(_rView) as DisplayObjectContainer).getChildAt(0);
            if (isNaN(_drawnHeight)) {
                _drawnHeight = _drawn.height;
                _drawnWidth = _drawn.width;
            }
            _bmd = new BitmapData(_drawnWidth, _drawnHeight, true, 0x00ffffff);
            _matrix.createBox(1, -1, 0, _drawnWidth/2, _drawnHeight/2);
            _bmd.draw(_drawn, _matrix);
            (_reflectPlane.material as BitmapMaterial).bitmap = _bmd;
            
            // find 2d bounding box of cube and redraw reflection
            _drawn = (_rView2.session.getContainer(_rView2) as DisplayObjectContainer).getChildAt(0);
            if (isNaN(_drawnHeight2)) {
                _drawnHeight2 = _drawn.height;
                _drawnWidth2 = _drawn.width;
            }
            _bmd = new BitmapData(_drawnWidth2, _drawnHeight2, true, 0x00ffffff);
            _matrix.createBox(1, -1, 0, _drawnWidth2/2, _drawnHeight2/2);
            _bmd.draw(_drawn, _matrix);
            (_reflectPlane2.material as BitmapMaterial).bitmap = _bmd;
            
            if (_sphere.y > _cube.y) {
                _reflectPlane.pushback = true;
                _reflectPlane2.pushback = true;
            } else {
                _reflectPlane2.pushback = true;
                _reflectPlane.pushback = true;
            }
            
            // render main view
            _view.render();
        }
    }
}
