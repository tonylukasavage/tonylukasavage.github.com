package com.transmote.flar.utils.geom {
    import jp.nyatla.nyartoolkit.as3.core.types.matrix.NyARDoubleMatrix34;
    
    import org.papervision3d.core.math.Matrix3D;
    
    /**
     * @author    Eric Socolofsky
     * @url        http://transmote.com/flar
     */
    public class FLARPVGeomUtils {
        
        /**
         * convert a FLARToolkit matrix to a Papvervision3D matrix.
         * @param    fm            FLARToolkit FLARDoubleMatrix34 to convert.
         * @param    bMirror        if true, this method will flip the resultant matrix horizontally (along the y-axis).
         * @return                Papervision Matrix3D generated from the FLARToolkit matrix.
         */
        public static function convertFLARMatrixToPVMatrix (fm:NyARDoubleMatrix34, bMirror:Boolean=true) :Matrix3D {
            if (bMirror) {
                return new org.papervision3d.core.math.Matrix3D ([
                    -fm.m01,    fm.m00,        -fm.m02,    -fm.m03,
                    -fm.m11,    fm.m10,        -fm.m12,    -fm.m13,
                    fm.m21,        -fm.m20,    fm.m22,        fm.m23
                    ]);
            } else {
                return new org.papervision3d.core.math.Matrix3D ([
                    fm.m01,        fm.m00,        fm.m02,        fm.m03,
                    -fm.m11,    -fm.m10,    -fm.m12,    -fm.m13,
                    fm.m21,        fm.m20,        fm.m22,        fm.m23
                    ]);
            }
        }
        
        public function FLARPVGeomUtils () {}
    }
}
