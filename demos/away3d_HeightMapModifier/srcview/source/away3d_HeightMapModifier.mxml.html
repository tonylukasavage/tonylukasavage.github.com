<?xml version="1.0" encoding="utf-8"?>
<s:Application xmlns:fx="http://ns.adobe.com/mxml/2009" 
               xmlns:s="library://ns.adobe.com/flex/spark" 
               xmlns:mx="library://ns.adobe.com/flex/mx" 
               width="800" height="600" frameRate="60"
               creationComplete="_onCreationComplete();" xmlns:local="*" viewSourceURL="srcview/index.html">
    <fx:Script>
        <![CDATA[
            import away3d.cameras.HoverCamera3D;
            import away3d.containers.View3D;
            import away3d.core.utils.Cast;
            import away3d.materials.BitmapMaterial;
            import away3d.materials.EnviroBitmapMaterial;
            import away3d.materials.WireColorMaterial;
            import away3d.modifiers.HeightMapModifier;
            import away3d.primitives.Skybox;
            import away3d.primitives.Sphere;
            
            import flash.display.Bitmap;
            import flash.display.BitmapData;
            import flash.display.Sprite;
            import flash.display.StageQuality;
            import flash.events.Event;
            import flash.filters.BevelFilter;
            import flash.geom.Matrix;
            
            import mx.core.UIComponent;
            
            private var HEIGHTMAP_WIDTH:Number = 256;
            private var HEIGHTMAP_HEIGHT:Number = 256;
            private var NUMBER_OF_SUBDIVISIONS:Number = 15;
            private var ELEVATION_HEIGHT:Number = 1;
            
            [Embed("resource/skybox/left.jpg")] private var leftImage:Class;
            [Embed("resource/skybox/right.jpg")] private var rightImage:Class;
            [Embed("resource/skybox/top.jpg")] private var topImage:Class;
            [Embed("resource/skybox/bottom.jpg")] private var bottomImage:Class;
            [Embed("resource/skybox/front.jpg")] private var frontImage:Class;
            [Embed("resource/skybox/back.jpg")] private var backImage:Class;
            
            [Embed("resource/hm/hm1.jpg")] private var hm1Image:Class;
            [Embed("resource/hm/hm2.jpg")] private var hm2Image:Class;
            [Embed("resource/hm/hm3.jpg")] private var hm3Image:Class;
            [Embed("resource/hm/hm4.jpg")] private var hm4Image:Class;
            
            [Embed("resource/enviro2.jpg")] private var enviroImage:Class;
            [Embed("resource/chrome.jpg")] private var testImage:Class;
            private var _hm1Bitmap:Bitmap = new Bitmap(Cast.bitmap(hm1Image));
            private var _hm2Bitmap:Bitmap = new Bitmap(Cast.bitmap(hm2Image));
            private var _hm3Bitmap:Bitmap = new Bitmap(Cast.bitmap(hm3Image));
            private var _hm4Bitmap:Bitmap = new Bitmap(Cast.bitmap(hm4Image));

            private var _camera:HoverCamera3D;
            private var _heightData:BitmapData = new BitmapData(HEIGHTMAP_WIDTH,HEIGHTMAP_HEIGHT);
            private var _bmDisplay:Bitmap;            
            private var _overlayAlpha:Number = 1.0;
            private var _sphere:Sphere;
            private var _heightMapModifier:HeightMapModifier;

            private var heightMapSprite:Sprite = new Sprite();
            private var matrix:Matrix = new Matrix();
            private var wireMaterial:WireColorMaterial = new WireColorMaterial(0x000000, {wireColor:0xffffff});
            private var enviroMaterial:EnviroBitmapMaterial = new EnviroBitmapMaterial(Cast.bitmap(testImage), Cast.bitmap(enviroImage));
            private var _changedMesh:Boolean = false;
            
            private function _onCreationComplete():void {
                if (stage) {
                    _init();
                } else {
                    this.addEventListener(Event.ADDED_TO_STAGE, function(e:Event):void { _init(); } );
                }
            }
            
            private function _init():void {
                var ui:UIComponent = new UIComponent();
                stage.quality = flash.display.StageQuality.LOW;
                
                // set some initial values
                enviroMaterial.reflectiveness = 0.5;
                divisionSlider.value = NUMBER_OF_SUBDIVISIONS;
                elevationSlider.value = ELEVATION_HEIGHT;
                
                // setup view
                away3dMain.title = "";
                _camera = new HoverCamera3D({panAngle:0, tiltAngle:0,  distance:2000});
                away3dMain.view.camera = _camera;
                _camera.hover(true);
                ui.addChild(away3dMain.view);
                this.addElementAt(ui,0);
                
                // setup height maps
                _heightData = new BitmapData(HEIGHTMAP_WIDTH, HEIGHTMAP_HEIGHT);
                heightMapSprite = new Sprite();
                heightMapSprite.addChild(_hm1Bitmap);
                heightMapSprite.addChild(_hm2Bitmap);
                heightMapSprite.addChild(_hm3Bitmap);
                heightMapSprite.addChild(_hm4Bitmap);
                _heightData.draw(heightMapSprite);
                
                // create skybox
                var skybox:Skybox = new Skybox(new BitmapMaterial(Cast.bitmap(backImage)),
                    new BitmapMaterial(Cast.bitmap(leftImage)),
                    new BitmapMaterial(Cast.bitmap(frontImage)),
                    new BitmapMaterial(Cast.bitmap(rightImage)),
                    new BitmapMaterial(Cast.bitmap(topImage)),
                    new BitmapMaterial(Cast.bitmap(bottomImage)));
                away3dMain.view.scene.addChild(skybox);
                
                // create the spehere that will be morphed by the HeightMapModifier
                _sphere = new Sphere({segmentsW:NUMBER_OF_SUBDIVISIONS, segmentsH:NUMBER_OF_SUBDIVISIONS, material:enviroMaterial});
                _sphere.bothsides = true;
                away3dMain.view.scene.addChild(_sphere);
                _camera.target = _sphere;
                
                // Create the HeightMapModifier
                _heightMapModifier = new HeightMapModifier(_sphere, _heightData);
                _heightMapModifier.offset = 300;
                _heightMapModifier.scale = ELEVATION_HEIGHT;
                
                // Show the current height map being applied to the sphere 
                _bmDisplay = new Bitmap();
                _bmDisplay.x = 700;
                _bmDisplay.scaleX = 100.0 / Number(HEIGHTMAP_WIDTH);
                _bmDisplay.scaleY = 100.0 / Number(HEIGHTMAP_HEIGHT);
                _bmDisplay.filters = [new BevelFilter()];
                ui.addChild(_bmDisplay);

                // pre-render processing
                away3dMain.preRender = function():void { 
                    _camera.hover();
                    _overlayAlpha -= 0.02;
                    
                    // change the overlay alpha for transitioning and create new height map
                    var child:Bitmap = heightMapSprite.getChildAt(heightMapSprite.numChildren-1) as Bitmap;
                    child.alpha = _overlayAlpha;
                    if (child.alpha <= 0) {
                        heightMapSprite.setChildIndex(child, 0);
                        child.alpha = 1;
                        _overlayAlpha = 2;
                    }
                    _heightData.draw(heightMapSprite);
                    _bmDisplay.bitmapData = _heightData;
                    
                    if (_changedMesh) {
                        away3dMain.view.scene.removeChild(_sphere);
                        _sphere = new Sphere({segmentsW:NUMBER_OF_SUBDIVISIONS, segmentsH:NUMBER_OF_SUBDIVISIONS, 
                            material:(wfCheck.selected ? wireMaterial : enviroMaterial) });
                        _sphere.bothsides = true;
                        away3dMain.view.scene.addChild(_sphere);
                        
                        _heightMapModifier = new HeightMapModifier(_sphere, _heightData);
                        _heightMapModifier.offset = 300;
                        _heightMapModifier.scale = ELEVATION_HEIGHT;
                        _changedMesh = false;
                    }
                    
                    _heightMapModifier.execute();
                    
                };
                away3dMain.postRender = function():void { trace("postRender"); };
                
                stage.addEventListener(MouseEvent.MOUSE_MOVE, _onMouseMove);
            }
            
            private function _onMouseMove(e:MouseEvent):void {
                _camera.panAngle = (e.stageX-(stage.stageWidth/2)) * 90 / (stage.stageWidth/2);
                _camera.tiltAngle = (e.stageY-(stage.stageHeight/2)) * 30 / (stage.stageHeight/2);
            }

            protected function checkbox1_clickHandler(event:MouseEvent):void
            {
                var cb:CheckBox = event.target as CheckBox;
                if (cb.selected) {
                    _sphere.material = wireMaterial;
                    reflectiveLabel.visible = false;
                    reflectiveSlider.visible = false;
                } else {
                    _sphere.material = enviroMaterial;
                    reflectiveLabel.visible = true;
                    reflectiveSlider.visible = true;
                }
            }

            protected function elevationSlider_changeHandler(event:Event):void
            {
                var hs:HSlider = event.target as HSlider;
                ELEVATION_HEIGHT = hs.value;
                _changedMesh = true;
            }


            protected function divisionSlider_changeHandler(event:Event):void
            {
                var hs:HSlider = event.target as HSlider;
                NUMBER_OF_SUBDIVISIONS = hs.value;
                _changedMesh = true;
            }
            
            protected function reflectiveSlider_changeHandler(event:Event):void
            {
                var hs:HSlider = event.target as HSlider;
                enviroMaterial.reflectiveness = Number(hs.value)/10.0;
            }

        ]]>
    </fx:Script>
    <local:Away3dUIComponent id="away3dMain" chromeColor="#000000"/>
    <s:CheckBox x="156" y="10" label="Show Wireframe" click="checkbox1_clickHandler(event)" color="#FFFFFF" id="wfCheck"/>
    <s:HSlider x="476" y="17" width="177" id="elevationSlider" minimum="-0.5" maximum="4" stepSize="0.5" value="1" change="elevationSlider_changeHandler(event)"/>
    <s:HSlider x="477" y="36" width="177" id="divisionSlider" minimum="5" maximum="30" stepSize="5" value="15" change="divisionSlider_changeHandler(event)"/>
    <s:HSlider x="477" y="55" width="177" id="reflectiveSlider" minimum="0" maximum="10" stepSize="1" value="5" change="reflectiveSlider_changeHandler(event)"/>
    <s:Label x="355" y="17" text="Elevation Height" width="113" color="#000000"/>
    <s:Label x="355" y="37" text="# of Subdivisions" width="114" color="#000000"/>
    <s:Label x="700" y="110" text="Height Map" width="100" textAlign="center" color="#FFFFFF" fontWeight="bold"/>
    <s:Label x="357" y="55" text="Reflectiveness" width="114" color="#000000" id="reflectiveLabel"/>
