package 
{
    import away3d.containers.ObjectContainer3D;
    import away3d.core.utils.Init;
    import away3d.primitives.Plane;

    public class CarouselGallery extends ObjectContainer3D
    {
        private var _noOfPlanes:int;
        private var _planeSpacing:Number;
        private var _planeHeight:Number;
        private var _planeWidth:Number;
        private var _galleryMaterials:Array;
        
        public function CarouselGallery(init:Object=null) {
            super();
            
            _galleryMaterials = new Array();
            
            var ini:Init = Init.parse(init);
            _planeHeight = ini.getNumber("planeHeight", 50);
            _planeWidth = ini.getNumber("planeWidth", 50);
            _planeSpacing = ini.getNumber("planeSpacing", _planeWidth/10);
            _galleryMaterials = ini.getArray("galleryMaterials");
            _noOfPlanes = _galleryMaterials.length;
            
              _drawPlanes(true);
        }
        
        public function set planeWidth(value:Number):void {
            _planeWidth = value;
            _drawPlanes(false);
        }
        
        public function get planeWidth():Number {
            return _planeWidth;
        }
        
        public function set planeHeight(value:Number):void {
            _planeHeight = value;
            _drawPlanes();
        }
        
        public function get planeHeight():Number {
            return _planeHeight;
        }
        
        public function set planeSpacing(value:Number):void {
            _planeSpacing = value;
            _drawPlanes();
        }
        
        public function get planeSpacing():Number {
            return _planeSpacing;
        }
        
        private function _drawPlanes(create:Boolean=false):void {
            var plane:Plane;
            var interiorAngle:Number = (_noOfPlanes - 2) * 180 / _noOfPlanes;
            var rotAngle:Number = 180 - interiorAngle;
            var baseAngle:Number = 360 / _noOfPlanes;
            var distance:Number = _planeWidth * Math.tan(_toRadians(interiorAngle/2)) / 2 + _planeSpacing;
            var totalBaseAngle:Number = 0;
            var totalRotAngle:Number = 90;
            var planes:Array = new Array();
            var i:int;
            
            if (create) {
                for (i = 0; i < _noOfPlanes; i++) {
                    plane = new Plane({material:_galleryMaterials[i]});
                    this.addChild(plane);
                    planes.push(plane);
                }
            } else {
                for (i = 0; i < this.children.length; i++) {
                    planes.push(this.children[i] as Plane);
                }
            }
            
            
            for (i = 0; i < _noOfPlanes; i++) {
                plane = planes[i] as Plane;
                this.addChild(plane);
                plane.bothsides = true;
                plane.rotationX = 90;
                plane.rotationY = totalRotAngle;
                plane.x = Math.cos(_toRadians(totalBaseAngle)) * distance;
                plane.y = 0;
                plane.z = Math.sin(_toRadians(totalBaseAngle)) * distance;;
                plane.width = _planeWidth;
                plane.height = _planeHeight;
                
                // get next values
                totalRotAngle -= rotAngle;
                totalBaseAngle += baseAngle;
            }
        }
        
        private function _toRadians(degrees:Number):Number {
            return degrees * Math.PI / 180;
        }
    }
}
