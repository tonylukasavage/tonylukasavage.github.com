<?xml version="1.0" encoding="utf-8"?>
<s:Application xmlns:fx="http://ns.adobe.com/mxml/2009" 
               xmlns:s="library://ns.adobe.com/flex/spark" 
               xmlns:mx="library://ns.adobe.com/flex/mx" 
               width="800" height="600" xmlns:local="*"
               backgroundColor="0x000000" frameRate="60"
               creationComplete="_onCreationComplete();" viewSourceURL="srcview/index.html">
    <fx:Script>
        <![CDATA[
            import away3d.cameras.HoverCamera3D;
            import away3d.core.session.BitmapSession;
            import away3d.core.utils.Cast;
            import away3d.materials.BitmapMaterial;
            import away3d.primitives.Cube;
            import away3d.primitives.Plane;
            
            import com.greensock.*;
            import com.greensock.easing.*;
            
            import mx.core.UIComponent;
            
            import spark.components.HSlider;
            
            [Embed("table.jpg")] private var tableImage:Class;
            [Embed("homebar.jpg")] private var barImage:Class;
            
            private var _camera:HoverCamera3D;
            private var _uiScene:UIComponent;
            private var _slider:HSlider;
            private var _lastValue:int = 10;
            
            private function _onCreationComplete():void {
                if (stage) {
                    _init();
                } else {
                    this.addEventListener(Event.ADDED_TO_STAGE, function(e:Event):void { _init(); } );
                }
            }
    
            private function _init():void {
                _uiScene = new UIComponent();
                stage.quality = flash.display.StageQuality.LOW;
                
                // setup view
                away3dMain.title = "";
                _camera = new HoverCamera3D({panAngle:0, tiltAngle:0, minTiltAngle:0, distance:1000});
                _camera.hover(true);
                away3dMain.view.camera = _camera;
                away3dMain.view.session = new BitmapSession(1);
                away3dMain.view.visible = false;
                this.addElement(_uiScene);
                
                // add some wine bottles
                var bottle:WineBottle = new WineBottle();
                bottle.rotationX = 90;
                bottle.rotationY = -90;
                away3dMain.view.scene.addChild(bottle);
                
                var bottle2:WineBottle = new WineBottle();
                bottle2.applyPosition((bottle2.minX+bottle2.maxX)/2, (bottle2.minY+bottle2.maxY)/2, (bottle2.minZ+bottle2.maxZ)/2);
                bottle2.rotationY = 90;
                bottle2.roll(90);
                bottle2.applyRotations();
                bottle2.rotationX = -15;
                bottle2.x = 200;
                bottle2.y = -bottle2.objectDepth/2 + bottle2.objectHeight/2;
                bottle2.z = 125;
                away3dMain.view.scene.addChild(bottle2);
                TweenMax.to(bottle2, 2, {z:100, rotationX:15, yoyo:true, repeat:-1, ease:Linear.easeInOut});

                var bottle3:WineBottle = new WineBottle();
                bottle3.x = -150;
                bottle3.y = -bottle3.objectDepth/2 + bottle3.objectHeight/2;
                bottle3.z = bottle3.objectDepth/2;
                bottle3.rotationX = 0;
                bottle3.rotationY = -45;
                away3dMain.view.scene.addChild(bottle3);
                
                // add the stand for the bottles
                var stand:Cube = new Cube({material:new BitmapMaterial(Cast.bitmap(tableImage))});
                stand.width = 3000;
                stand.depth = 1000;
                stand.height = 500;
                stand.y = -stand.height/2 - bottle.objectDepth/2;
                stand.z = 0;
                stand.segmentsD = 4;
                stand.segmentsW = 4;
                stand.segmentsH = 4;
                stand.ownCanvas = true;
                stand.pushback = true;
                away3dMain.view.scene.addChild(stand);

                // add the bar background
                var background:Plane = new Plane({height:3000, width:3000, material:new BitmapMaterial(Cast.bitmap(barImage))});
                background.z = -500;
                background.rotationX = -90;
                background.ownCanvas = true;
                background.pushback = true;
                away3dMain.view.scene.addChild(background);

                // handle the text and background for our slider
                var ui:UIComponent = new UIComponent();
                var block:Sprite = new Sprite();
                block.graphics.beginFill(0x000000);
                block.graphics.drawRect(0,0,800,50);
                block.graphics.endFill();
                ui.addChild(block);
                
                var tf:TextFormat = new TextFormat("arial", 16, 0xffffff, true);
                var sober:TextField = new TextField();
                sober.defaultTextFormat = tf;
                sober.text = "SOBER";
                sober.x = -110;
                sober.y = 16;
                sober.width = 60;
                ui.addChild(sober);
                TweenLite.to(sober, 2, {x:10});
                
                var wasted:TextField = new TextField();
                wasted.defaultTextFormat = tf;
                wasted.text = "WASTED";
                wasted.x = 815;
                wasted.y = 16;
                wasted.width = 80;
                ui.addChild(wasted);
                TweenLite.to(wasted, 2, {x:715});
                this.addElement(ui);
                
                // add sobriety slider
                _slider = new HSlider();
                _slider.x = 75;
                _slider.y = 20;
                _slider.width = 630;
                _slider.minimum = 5;
                _slider.maximum = 20;
                _slider.stepSize = 1;
                _slider.value = 10;
                this.addElement(_slider);
                
                away3dMain.preRender = function():void {
                    _camera.hover();
                }
                
                away3dMain.postRender = function():void {
                    if (_slider.value < 6) {
                        while (_uiScene.numChildren) { _uiScene.removeChildAt(0); }
                    } else if (_lastValue < 6) {
                        _uiScene.removeChildAt(0);
                    }
                    
                    var bitmap:Bitmap = new Bitmap(away3dMain.view.getBitmapData().clone());
                    bitmap.alpha = 1;
                    bitmap.cacheAsBitmap = true;
                    _uiScene.addChildAt(bitmap,0);
                    
                    if (_slider.value > 5) {
                        TweenLite.to(bitmap, _slider.value * 0.05, {alpha:0, ease:Linear.easeNone, onCompleteParams:[bitmap], 
                            onComplete:function(bm:Bitmap):void {
                                if (_uiScene.contains(bm)) {
                                    _uiScene.removeChild(bm);
                                }
                                bm = null;
                            }
                        });
                    } 
                    _lastValue = _slider.value;
                }
                stage.addEventListener(MouseEvent.MOUSE_MOVE, _onMouseMove);    
            }
            
            private function _onMouseMove(e:MouseEvent):void {
                _camera.panAngle = (e.stageX-(stage.stageWidth/2)) * 30 / (stage.stageWidth/2);
                _camera.tiltAngle = (stage.stageHeight - e.stageY) * 15 / stage.stageHeight;
            }

        ]]>
    </fx:Script>
    <local:Away3dUIComponent id="away3dMain" height="100%" width="100%" y="50"/>
</s:Application>
