<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" layout="absolute" xmlns:local="*"
                backgroundColor="0xbbbbbb" creationComplete="init();" frameRate="30" height="600" width="800" viewSourceURL="srcview/index.html">
    <mx:Script>
        <![CDATA[
            import away3dlite.containers.ObjectContainer3D;
            import away3dlite.core.base.Face;
            import away3dlite.materials.WireColorMaterial;
            import __AS3__.vec.Vector;
            import away3dlite.primitives.Sphere;
            import away3dlite.primitives.Plane;
            
            private var source:Sphere;
            private var planes:Vector.<Plane> = new Vector.<Plane>();
            private var sourceSegmentsChange:Boolean = false;
            private var sourceSize:Number = 25;
            private var container:ObjectContainer3D
            
            private function init():void {
                if (stage) {
                    initAway3D();
                } else {
                    this.addEventListener(Event.ADDED_TO_STAGE, function(e:Event):void { stage.quality = flash.display.StageQuality.LOW; 
                                                                                         initAway3D(); });
                }    
            }    
            
            private function initAway3D():void {
                away3dMain.view.camera.z = -100;
                away3dMain.view.camera.lookAt(new Vector3D(0,0,0));    
                
                container = new ObjectContainer3D();
                away3dMain.view.scene.addChild(container);
                createSourceSphere();
                createLinks();
                
                this.addEventListener(Event.ENTER_FRAME, render);
            }
            
            private function createLinks():void {
                var i:uint;
                
                for (i = 0; i < planes.length; i++) {
                    container.removeChild(planes[i]);
                }
                planes.length = 0;
                
                for (i = 0; i < source.faces.length; i++) {
                    var plane:Plane = new Plane(new WireColorMaterial(0x000000, 1, 0xffffff), 10, 10, 1, 1, false);
                    plane.bothsides = true;
                    plane.mouseChildren = false;
                    plane.mouseEnabled = false;
                    planes.push(plane);
                    container.addChild(planes[i]); 
                }
            }
            
            private function _rotate(rot:Vector3D, obj:Vector3D):Vector3D {
                var _rad:Number = Math.PI / 180;
                var rotx:Number = rot.x * _rad;
                var roty:Number = rot.y * _rad;
                var rotz:Number = rot.z * _rad;
                var sinx:Number = Math.sin(rotx);
                var cosx:Number = Math.cos(rotx);
                var siny:Number = Math.sin(roty);
                var cosy:Number = Math.cos(roty);
                var sinz:Number = Math.sin(rotz);
                var cosz:Number = Math.cos(rotz);
     
                var x:Number = obj.x;
                var y:Number = obj.y;
                var z:Number = obj.z;
    
                var y1:Number = y;
                y = y1*cosx+z*-sinx;
                z = y1*sinx+z*cosx;
                
                var x1:Number = x;
                x = x1*cosy+z*siny;
                z = x1*-siny+z*cosy;
            
                x1 = x;
                x = x1*cosz+y*-sinz;
                y = x1 * sinz + y * cosz;
                
                return new Vector3D(x,y,z);
            }
            
            private function _updateFaceLinks():void {
                var offset:Number = offsetSlider.value;
                var face:Face, plane:Plane;
                var rot:Vector3D = new Vector3D(source.rotationX, source.rotationY, source.rotationZ);
                var center:Vector3D, normal:Vector3D;
                
                for (var i:uint = 0; i < source.faces.length; i++) {
                    face = source.faces[i] as Face;
                    plane = planes[i];
                    center = _rotate(rot, face.center);
                    normal = _rotate(rot, face.normal);
    
                    plane.x = center.x + (offset * normal.x);
                    plane.y = center.y + (offset * normal.y);
                    plane.z = center.z + (offset * normal.z);
                    plane.lookAt(new Vector3D(center.x, center.y, center.z));
                }
            }
            
            private function onShowChange(event:Event):void {
                if (checkMesh.selected) {
                    source.visible = true;
                } else {
                    source.visible = false;
                }
            }
            
            private function changeSourceSegments():void {
                sourceSegmentsChange = true;
            }
            
            private function createSourceSphere():void {
                if (source && container.contains(source)) { container.removeChild(source); }
                source = new Sphere(new WireColorMaterial(0xffffff, 1, 0x000000), sourceSize, sourceSlider.value, sourceSlider.value);
                source.visible = checkMesh.selected;
                source.mouseEnabled = false;
                source.mouseChildren = false;
                container.addChild(source);
            }
            
            private function render(evt:Event):void {
                if (away3dMain.doRender) { 
                    container.rotationX += 0.5;
                    container.rotationY += 0.5;
                    container.rotationZ += 0.5;

                    if (sourceSegmentsChange) {
                        sourceSegmentsChange = false;
                        createSourceSphere();
                        createLinks();
                    }
                    
                    _updateFaceLinks();
                    away3dMain.view.render(); 
                } 
            }

                labels="[1,50,100]" liveDragging="false" id="offsetSlider" value="10"/>
